\chapter{Introduction}
\begin{Definition}
  {Web Application}
  A web application is a graphical computer program that the user interacts with in a web browser. They often also have a server-side component that runs on a webserver

  Examples: 
  \begin{itemize}
    \item Gmail
    \item Facebook
    \item Twitter
    \item Discord
    \item Reddit
    \item etc
  \end{itemize}
  
\end{Definition}
